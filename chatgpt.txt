    public function index(Request $req)
    {
        $q = Payment::query()
            ->when($req->status, fn($qq)=>$qq->where('status',$req->status))
            ->when($req->from,   fn($qq)=>$qq->whereDate('created_at','>=',$req->from))
            ->when($req->to,     fn($qq)=>$qq->whereDate('created_at','<=',$req->to))
            ->orderByDesc('id')
            ->paginate(20);
